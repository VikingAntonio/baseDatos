-- SQL para las tablas de Visual DB Designer

CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- Tabla bdd_users: Almacena la información de los usuarios
CREATE TABLE bdd_users (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    username TEXT UNIQUE NOT NULL,
    password TEXT NOT NULL, -- Las contraseñas se almacenan como hashes SHA-256
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Tabla projects: Almacena los proyectos vinculados a cada usuario
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES bdd_users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    data JSONB NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(user_id, name)
);
